Create Lambda Function to Process CSV Files

Objectives

  • Automatically trigger when a new CSV file is uploaded to S3.
  • Read each row in the CSV, call Comprehend DetectSentiment to analyze the sentiment of the review.
  • Generate a result CSV file and save it in the processed/ folder in the same or another bucket.

Step 1: Create Lambda Function

  1. Go to AWS Lambda Console → Create function.

  2. Choose Author from scratch:

    • Name: ComprehendCSVProcessor
    • Runtime: Python 3.13 (or the latest version)
    • Role: Select NLP-Lambda-Role created in section 2.4.
    • Click Create function.

Step 2: Lambda Code

  1. Paste the following code into the Lambda Code source:
import boto3
import csv
import io
import json
import os
import urllib.parse
from datetime import datetime

# CONFIG - change according to your environment
REGION = "ap-southeast-1"  
LANGUAGE_CODE = "en"        
SENSITIVE_WORDS = {         # sample list; can be loaded from S3 if you need a larger list
    "damn", "shit", "trash", "bastard", "f**k", "m*"
}

# Initialize clients
s3 = boto3.client("s3", region_name=REGION)
comprehend = boto3.client("comprehend", region_name=REGION)

# Map sentiment string -> numeric label (used for training)
SENTIMENT_TO_LABEL = {
    "POSITIVE": 0,
    "NEGATIVE": 1,
    "NEUTRAL": 2,
    "MIXED": 3,
    "N/A": 4
}


def normalize_token(t: str):
    return t.strip().lower()


def analyze_text(text: str):
    """
    Call Comprehend APIs and return a detailed result dictionary.
    """
    if not text or text.strip() == "":
        return {
            "sentiment": "N/A",
            "sentiment_scores": {"Positive": 0, "Negative": 0, "Neutral": 0, "Mixed": 0},
            "key_phrases": [],
            "entities": [],
            "syntax_tokens": []
        }

    sent = comprehend.detect_sentiment(Text=text, LanguageCode=LANGUAGE_CODE)
    kp = comprehend.detect_key_phrases(Text=text, LanguageCode=LANGUAGE_CODE)
    ents = comprehend.detect_entities(Text=text, LanguageCode=LANGUAGE_CODE)
    syntax = comprehend.detect_syntax(Text=text, LanguageCode=LANGUAGE_CODE)

    return {
        "sentiment": sent.get("Sentiment"),
        "sentiment_scores": sent.get("SentimentScore", {}),
        "key_phrases": kp.get("KeyPhrases", []),
        "entities": ents.get("Entities", []),
        "syntax_tokens": syntax.get("SyntaxTokens", [])
    }


def mark_sensitive_tokens(syntax_tokens):
    """
    From Comprehend syntax tokens, mark tokens found in SENSITIVE_WORDS.
    Returns a list of token dicts:
    {"text":..., "beginOffset":..., "part_of_speech":..., "is_sensitive": True/False}
    """
    out = []
    for tk in syntax_tokens:
        txt = tk.get("Text", "")
        pos_tag = tk.get("PartOfSpeech", {}).get("Tag")
        is_sens = normalize_token(txt) in SENSITIVE_WORDS
        out.append({
            "text": txt,
            "beginOffset": tk.get("BeginOffset"),
            "part_of_speech": pos_tag,
            "is_sensitive": is_sens
        })
    return out


def lambda_handler(event, context):
    """
    Lambda entry point expects an S3 Put event.
    For manual testing, craft an event with 'bucket' and 'key'.
    """
    try:
        record = event["Records"][0]
        bucket = record["s3"]["bucket"]["name"]
        key = urllib.parse.unquote_plus(record["s3"]["object"]["key"])
    except Exception as e:
        return {"statusCode": 400, "body": f"Invalid event format: {e}"}

    base_name = os.path.basename(key)
    timestamp = datetime.utcnow().strftime("%Y%m%dT%H%M%SZ")

    # Temporary local paths
    download_path = f"/tmp/{base_name}"
    output_jsonl_path = f"/tmp/processed_{base_name.rsplit('.',1)[0]}_{timestamp}.jsonl"
    output_csv_path = f"/tmp/summary_{base_name.rsplit('.',1)[0]}_{timestamp}.csv"

    # Download CSV from S3
    try:
        s3.download_file(bucket, key, download_path)
    except Exception as e:
        return {"statusCode": 500, "body": f"Error downloading s3://{bucket}/{key} - {e}"}

    # Open CSV and process each row
    jsonl_file = open(output_jsonl_path, "w", encoding="utf-8")
    csv_buffer = io.StringIO()
    csv_writer = csv.writer(csv_buffer, quoting=csv.QUOTE_MINIMAL)
    csv_writer.writerow(["id", "customer", "review", "label", "label_str", "top_key_phrases", "sensitive_words"])

    processed_count = 0
    with open(download_path, newline="", encoding="utf-8") as csvfile:
        reader = csv.DictReader(csvfile)
        for idx, row in enumerate(reader):
            processed_count += 1
            doc_id = f"{base_name}_{idx}"
            customer = row.get("customer", "")
            review_text = row.get("review", "")

            # Call Comprehend
            result = analyze_text(review_text)

            # Token-level sensitive marking
            tokens = mark_sensitive_tokens(result.get("syntax_tokens", []))

            # Collect sensitive words found
            sensitive_found = sorted(list({t["text"] for t in tokens if t["is_sensitive"]}))

            # Collect top key phrases
            key_phrases_list = [kp.get("Text") for kp in result.get("key_phrases", [])]
            entities_list = [{"text": e.get("Text"), "type": e.get("Type")} for e in result.get("entities", [])]

            # Document-level label
            label_str = result.get("sentiment", "N/A")
            label_num = SENTIMENT_TO_LABEL.get(label_str, 4)

            # JSONL entry
            doc = {
                "id": doc_id,
                "customer": customer,
                "text": review_text,
                "label": label_num,
                "label_str": label_str,
                "sentiment_scores": result.get("sentiment_scores", {}),
                "key_phrases": key_phrases_list,
                "entities": entities_list,
                "tokens": tokens
            }
            jsonl_file.write(json.dumps(doc, ensure_ascii=False) + "\n")

            # CSV summary row
            csv_writer.writerow([
                doc_id,
                customer,
                review_text,
                label_num,
                label_str,
                "|".join(key_phrases_list),
                "|".join(sensitive_found)
            ])

    jsonl_file.close()

    # Save CSV file locally as well
    with open(output_csv_path, "w", encoding="utf-8", newline="") as f:
        f.write(csv_buffer.getvalue())

    # Upload outputs back to S3
    output_jsonl_key = f"processed/{os.path.basename(output_jsonl_path)}"
    output_csv_key = f"processed/{os.path.basename(output_csv_path)}"

    try:
        s3.upload_file(output_jsonl_path, bucket, output_jsonl_key)
        s3.upload_file(output_csv_path, bucket, output_csv_key)
    except Exception as e:
        return {"statusCode": 500, "body": f"Error uploading results to S3: {e}"}

    return {
        "statusCode": 200,
        "body": {
            "message": "Processing complete",
            "processed_count": processed_count,
            "jsonl_s3": f"s3://{bucket}/{output_jsonl_key}",
            "csv_s3": f"s3://{bucket}/{output_csv_key}"
        }
    }
  1. Select Deploy (or press Ctrl + Shift + U) to save and deploy the code.

Step 3: Create Test Event

  1. Open AWS Lambda Console → select the function ComprehendCSVProcessor.
  2. Click Test → choose Create new event.
  3. Choose Event template: search for "Amazon S3 Put" (or "S3 Put").
  4. Modify the JSON as below (replace with your bucket name and file key):
{
  "Records": [
    {
      "s3": {
        "bucket": {
          "name": "bucket-name"
        },
        "object": {
          "key": "reviews.csv"
        }
      }
    }
  ]
}

The “key” must exactly match the path and filename in S3 (e.g., “input/customer_reviews.csv”).
If the filename contains spaces or special characters, it will be encoded (space → %20). Please copy it exactly from the S3 console.

Step 4: Run Test and Check Results

  1. Click Test → Lambda will read the file from S3 and process it.
  2. After completion:
    • Open S3 Console.
    • Navigate to the processed/ folder to view the resulting CSV files.

Here, two files are output: a JSONL file for training and a CSV file for flexible use.
